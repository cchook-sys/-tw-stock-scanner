50 : (bars[i].close - low) / (hi - low) * 100; } k = 2 / 3 * k + 1 / 3 * rsv; d = 2 / 3 * d + 1 / 3 * k; K.push(k); D.push(d); } return { K, D }; } function rsi(closes, n = 14) { const out = [50]; let ag = 0, al = 0; for (let i = 1; i < closes.length; i++) { const ch = closes[i] - closes[i - 1]; const g = Math.max(ch, 0), l = Math.max(-ch, 0); if (i <= n) { ag += g / n; al += l / n; } else { ag = (ag * (n - 1) + g) / n; al = (al * (n - 1) + l) / n; } out.push(al === 0 ? 100 : 100 - 100 / (1 + ag / al)); } return out; } function adLine(bars) { const out = []; let cum = 0; for (const b of bars) { const rng = b.high - b.low; const mfm = rng === 0 ? 0 : ((b.close - b.low) - (b.high - b.close)) / rng; cum += mfm * b.volume; out.push(cum); } return out; } function quantile(arr, q) { const s = [...arr].filter(v => v != null).sort((a, b) => a - b); if (!s.length) return null; const pos = (s.length - 1) * q, base = Math.floor(pos), rest = pos - base; return s[base + 1] !== undefined ? "+" : ""}${mom.toFixed(2)}`], ]; }, }, oversold: { name: "超跌反彈", icon: "🔄", kind: "逆勢", P: { LOW_LOOKBACK: 20, NEAR_LOW_PCT: 0.03, KD_OVERSOLD: 20, VOL_SHRINK: 0.8 }, scan(bars) { const P = this.P; if (bars.length < P.LOW_LOOKBACK + 5) return null; } function fmGet_(params) { // 單一請求（給 name map 等用） params.token = FINMIND_TOKEN; const q = Object.keys(params).map(k => k + '=' + encodeURIComponent(params[k])).join('&'); const resp = UrlFetchApp.fetch(FINMIND_URL + '?' + q, { muteHttpExceptions: true });
